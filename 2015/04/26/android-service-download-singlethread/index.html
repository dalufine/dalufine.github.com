android中Service断点下载-单线程

android
  • Service

下载并刷新UI的过程

下载的关键点

  • 获取网络文件的长度
  • 在本地创建一个文件，设置其长度
  • 从数据库中获取上次下载的进度
  • 从上次下载的位置开始下载数据，同时保存进度到数据库
  • 将下载进度回传给Activity
  • 下载完成后删除信息

项目结构

||-Activity

||-Service

-DownloadService(启动新线程获取文件信息FileInfo)
-DownloadTask(下载任务类，关联数据库，获取下载信息，开始下载任务)

||-Bean

-FileInfo（文件信息）
-ThreadInfo(线程下载信息：下载进度)

||-db

-DBHelper(创建数据库)
-ThreadDAO(线程的数据访问接口)
-ThreadDAOImpl(实现接口)

public void download(){
    //读取数据库的线程信息
    List<ThreadInfo> threadInfos = mThreadDAO.getThreads(mFileInfo.getUrl()) ;
    ThreadInfo threadInfo = null ;
    if(threadInfos.size() ==0 ){
        threadInfo  = new ThreadInfo(···) ;
    }else{
        //该例子中是单线程的，所以就一个
        threadInfo = threadInfos.get(0) ;
    }
    //创建子线程下载
    new DownloadThread(threadInfo).start() ;
}
//下载线程
class DownloadThread extends Thread{
    public void run(){
        URL url = new URL(fileUrl) ;
        HttpURLConnection conn = (HttpURLConnection)url.openConnection() ;
        conn.setConnectTimeout(3000);
        conn.setRequestMethod("GET");
        //1.设置下载位置
        //文件开始下载的位置
        int downStartPath = mThreadInfo.getStart()+mThreadInfo.getFinished() ;
        //这里设置的是Range，即部分下载，则下面的conn.getResponseCode() == HttpStatus.SC_OK==200就不能起作用了，应该设置为conn.getResponseCode() == HttpStatus.SC_PARTIAL_CONTENT==206
        conn.setRequestProperty("Range", "bytes="+downStartPath+"-"+mThreadInfo.getEnd());
        //2.设置写入位置
        File file = new File(DownloadService.DOWNLOAD_PATH,mFileInfo.getFileName()) ;
        RandomAccessFile raf - new RandomAccessFile(file, "rwd") ;
        //seek():在读写的时候，跳过设置的字节数downStartPath，从下一个字节开始读写
        raf.seek(downStartPath);
        Intent intent = new Intent(DownloadService.ACTION_UPDATE) ;
        //当前进度
        mFinish += mThreadInfo.getFinished() ;
        //3.开始下载
        if(conn.getResponseCode() == HttpStatus.SC_PARTIAL_CONTENT){
            long time = System.currentTimeMillis() ;
            //3.1读取数据
            InputStream input = conn.getInputStream() ;
            byte[] buffer = new byte[1024*4] ;
            int len = -1 ;
            while((len = input.read(buffer))!=-1){
                //3.2写入文件
                raf.write(buffer,0,len);
                mFinish += len ;
                //0.5s刷新一次
                if(System.currentTimeMillis()-time>500){
                    time = System.currentTimeMillis() ;
                    intent.putExtra("finished", mFinish*100/mFileInfo.getlength) ;
                    mContext.sendBroadcast(intent) ;
                }
                //3.3暂停时保存进度到数据库
                if(isPause){
                    mThreadDAO.updateThread(···) ;
                    return ;
                }
            }
            //4.删除线程信息
            mThreadDAO.deleteThread(···); 
        }
        //关闭流等
    }
}
